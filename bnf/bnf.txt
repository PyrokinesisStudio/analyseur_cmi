<value> ::= <t_identifier> | <t_integer> | <t_float>

<expr> ::= <value> <expr'> ;; <expr'>
<expr'> ::= <t_operator> <value> <expr'> | E ;; <t_operator> | E

<affe> ::= <t_identifier> "=" <expr> ";" ;; "="

<instr> ::= "{" <instr> "}" | <affe> ;; <instr> | <affe>

<cond> ::= "if" "(" <expr> ")" <instr>

<root> ::= <expr>
